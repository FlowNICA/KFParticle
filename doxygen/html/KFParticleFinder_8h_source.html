305  void CopyCuts(const KFParticleFinder* finder)
306  {
310  fDistanceCut = finder->fDistanceCut;
311  fLCut = finder->fLCut;
312  for(int iCut=0; iCut<3; iCut++)
313  fCuts2D[iCut] = finder->fCuts2D[iCut];
314  for(int iCut=0; iCut<3; iCut++)
315  fSecCuts[iCut] = finder->fSecCuts[iCut];
316  for(int iCut=0; iCut<3; iCut++)
317  for(int jCut=0; jCut<3; jCut++)
318  fCutsTrackV0[iCut][jCut] = finder->fCutsTrackV0[iCut][jCut];
319  for(int iCut=0; iCut<2; iCut++)
320  for(int jCut=0; jCut<3; jCut++)
321  fCutsPartPart[iCut][jCut] = finder->fCutsPartPart[iCut][jCut];
322  fCutCharmPt = finder->fCutCharmPt;
323  fCutCharmChiPrim = finder->fCutCharmChiPrim;
324  for(int iCut=0; iCut<3; iCut++)
325  fCutsCharm[iCut] = finder->fCutsCharm[iCut];
326  fCutLVMPt = finder->fCutLVMPt;
327  fCutLVMP = finder->fCutLVMP;
328  fCutJPsiPt = finder->fCutJPsiPt;
329  }
