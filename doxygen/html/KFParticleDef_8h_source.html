82  static inline __attribute__((always_inline)) float_v Sin ( const float_v &phi )
83  {
84  const float_v pi(3.1415926535897932f);
85  const float_v nTurnsF = (phi + pi) / (float_v(2.f)*pi);
86  int_v nTurns = simd_cast<int_v>( nTurnsF );
87  nTurns( (nTurns<=int_v(Vc::Zero)) && simd_cast<int_m>(phi<-pi)) -= 1;
88 
89  const float_v& x = phi - simd_cast<float_v>(nTurns)*(float_v(2.f)*pi);
90 
91  const float_v& B = 4.f/pi;
92  const float_v& C = -B/pi;
93 
94  float_v y = (B + C * Vc::abs(x)) * x;
95 
96  const float_v& P = 0.218f;
97  y = P * (y * Vc::abs(y) - y) + y;
98 
99  return y;
100  }

105  static inline __attribute__((always_inline)) float_v ATan2( const float_v &y, const float_v &x )
106  {
107  const float_v pi(3.1415926535897932f);
108  const float_v zero(Vc::Zero);
109 
110  const float_m &xZero = (x == zero);
111  const float_m &yZero = (y == zero);
112  const float_m &xNeg = (x < zero);
113  const float_m &yNeg = (y < zero);
114 
115  const float_v &absX = Vc::abs(x);
116  const float_v &absY = Vc::abs(y);
117 
118  float_v a = absY / absX;
119  const float_m &gt_tan_3pi_8 = (a > float_v(2.414213562373095f));
120  const float_m &gt_tan_pi_8 = (a > float_v(0.4142135623730950f)) && (!gt_tan_3pi_8);
121  float_v b(Vc::Zero);
122  b(gt_tan_3pi_8) = pi/2.f;
123  b(gt_tan_pi_8) = pi/4.f;
124  a(gt_tan_3pi_8) = (-1.f / a);
125  a(gt_tan_pi_8) = ((absY - absX) / (absY + absX)) ;
126  const float_v &a2 = a * a;
127  b += (((8.05374449538e-2f * a2
128  - 1.38776856032E-1f) * a2
129  + 1.99777106478E-1f) * a2
130  - 3.33329491539E-1f) * a2 * a
131  + a;
132  b(xNeg ^ yNeg) = -b;
133  b(xNeg && !yNeg) = (b+pi);
134  b(xNeg && yNeg) = (b-pi);
135  b(xZero && yZero) = zero;
136  b(xZero && yNeg) = (-pi/2.f);
137  return b;
138  }
