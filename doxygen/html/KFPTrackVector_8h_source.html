112  void RecalculateLastIndex()
113  {
115  fNE = 0; fNMu = 0; fNPi = 0; fNK = 0; fNP = 0; fND = 0; fNT = 0; fNHe3 = 0; fNHe4 = 0;
116  for(int i=0; i<Size(); i++)
117  {
118  switch (abs(fPDG[i]))
119  {
120  case 11: fNE++; break;
121  case 13: fNMu++; break;
122  case 19: fNMu++; break;
123  case 211: fNPi++; break;
124  case 1: fNPi++; break;
125  case 3: fNPi++; break;
126  case 321: fNK++; break;
127  case 2212: fNP++; break;
128  case 1000010020: fND++; break;
129  case 1000010030: fNT++; break;
130  case 1000020030: fNHe3++; break;
131  case 1000020040: fNHe4++; break;
132  default : std::cout << "WARNING! KFPTrackVector::RecalculateLastIndex() - Particle " << fPDG[i] << " is not found!" << std::endl;
133  }
134  }
135 
136  fNMu += fNE; fNPi += fNMu; fNK += fNPi; fNP += fNK;
137  fND += fNP; fNT += fND; fNHe3 += fNT; fNHe4 += fNHe3;
138  }
